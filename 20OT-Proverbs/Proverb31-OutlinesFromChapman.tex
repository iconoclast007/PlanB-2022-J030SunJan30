%\textbf{Introduction: }
\begin{compactenum}[I.]
    \item \textbf{The Compassion of Godly Mothers:}  \index[scripture]{Proverbs!Pro 31:11-28}(Proverbs 31:11-28)
    \begin{compactenum}[A.]
        \item Compassion for her family
        \item Compassion for her friends
        \item Compassion for her fellowship
        \begin{compactenum}[1.]
            \item She will love Christ
            \item She will be loyal to the church 
            \item She will labor for the cross
        \end{compactenum}
    \end{compactenum}
    \item \textbf{The Character of Godly Mothers:}  \index[scripture]{Proverbs!Pro 31:25}(Proverbs 31:25)
    \begin{compactenum}[A.]
        \item She has principles that are honest 
        \item She has praise that is honorable
        \item She has performance that is helpful
    \end{compactenum}
    \item \textbf{The Conviction of Godly Mothers:}  \index[scripture]{Proverbs!Pro 31:26}(Proverbs 31:26)
    \begin{compactenum}[A.]
        \item She displays a spirit of kindness
        \item She is dedicated to the Kingdom
        \item She delivers the word of Knowledge
    \end{compactenum}
    \item \textbf{The Courage of Godly Mothers:}  \index[scripture]{Proverbs!Pro 31:21}(Proverbs 31:21)
    \begin{compactenum}[A.]
        \item To witness for Christ 
        \item To warn her children 
        \item To work for her church
    \end{compactenum}
    \item \textbf{The Commitment of Godly Mothers:}  \index[scripture]{Proverbs!Pro 31:10-31}(Proverbs 31:10-31)
    \begin{compactenum}[A.]
        \item To serve Christ.
        \item To share Christ 
        \item To show Christ
    \end{compactenum}
\end{compactenum}
